Vivifyer.prototype._vivify = function (key) {
    var value = this._values[key]
    if (values == null) {
        values = this._values[key] = this._constructor.call(null, key)
    }

}

Vivifyer.prototype.get = function (key) {
    return this._vivify(key)
}
